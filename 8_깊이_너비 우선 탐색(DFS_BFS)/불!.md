# [백준] bfs - 불!
https://www.acmicpc.net/problem/4179

## 문제 접근 
> 지훈이랑 불이난 공간 큐에 넣고 지훈이 이동만 visited에 넣는다. visited는 범위+2해서 범위가 넘을떄 거리를 바로 출력할수 있도록 한다 -> 틀림 -> 이유: .으로 나가면서 탈출하는 경우가 아닌 그냥 범위를 넘었을 때를 출력해서 

## 코드(테케 맞고 & 7%에서 틀림)
```python
from collections import deque

r, c = map(int, input().split())
graph = []
for _ in range(r):
    graph.append(list(input()))
    
queue = deque()    

dx = [-1, 1, 0, 0]
dy = [0, 0, 1, -1]

visited = [[0]*(r+2) for _ in range(c+2)]

for i in range(r):
    for j in range(c):
        if graph[i][j] == 'J':
             queue.append((i, j))

for i in range(r):
    for j in range(c):
        if graph[i][j] == 'F':
            queue.append((i, j))

def bfs():
    while queue:
        x, y = queue.popleft()
        for i in range(4):
            nx = x+dx[i]
            ny = y+dy[i]
            if nx<0 or ny<0 or nx>=r or ny>= c:
                visited[nx+1][ny+1] = visited[x+1][y+1]+1
                return visited[nx+1][ny+1]
            if 0<=nx<r and 0<=ny<c and graph[nx][ny] == '.':
                if graph[x][y] == 'J':
                    graph[nx][ny] = 'J'
                    visited[nx+1][ny+1] = visited[x+1][y+1]+1
                    queue.append((nx, ny))           
                elif graph[x][y] == 'F':
                    graph[nx][ny] = 'F'
                    queue.append((nx, ny)) 
    return "IMPOSSIBLE"
print(bfs())
```
