# [백준] dfs/bfs - 양
https://www.acmicpc.net/problem/3184

## 문제 접근 방법
> 처음에 dfs로 풀었는데 테스트케이스등 정답은 맞는것같은데 런타임 에러가 난다. -> bfs로 풀어야됨
유기농배추(https://www.acmicpc.net/problem/1012)도 dfs로 풀어서 런타임 에러가 나서 bfs로 풀었더니 풀렸다. 
앞으로는 무조건 bfs로만 풀어야하는 습관을 기르자! 
풀이는 전형적인 bfs문제로 양과 늑대만 구별하면 된다. 

## 코드(내가 푼것)
```python
import sys 
input = sys.stdin.readline

r, c= map(int, input().split())
graph = []
for i in range(r):
    graph.append(list(input()))
    
sheep = 0
wolf = 0
def dfs(x, y):
    global sheep
    global wolf
    if x<=-1 or x>=r or y<=-1 or y>=c:
        return False
    if graph[x][y] != '#':
        if graph[x][y] == 'o':
            sheep+=1
        if graph[x][y] == 'v':
            wolf+=1
        graph[x][y] = '#'
        dfs(x-1, y)
        dfs(x, y-1)
        dfs(x+1, y)
        dfs(x, y+1)
        return True
    return False

sheep_count = 0
wolf_count=0
for i in range(r):
    for j in range(c):
        if dfs(i, j) == True
            if sheep>wolf and (sheep + wolf != 0):
                sheep_count+=sheep
            elif sheep<=wolf and (sheep + wolf != 0):
                wolf_count+=wolf
            sheep = 0
            wolf = 0
           
print(sheep_count, wolf_count)
```

## 코드(교재 풀이)
```python
from collections import deque
import sys 
input = sys.stdin.readline

r, c= map(int, input().split())
graph = []
for i in range(r):
    graph.append(list(input()))
    
dx = [-1, 1, 0, 0]
dy = [0, 0, -1, 1]

sheep_count = 0
wolf_count = 0

def bfs(x, y):
    global sheep_count
    global wolf_count
    
    queue = deque()
    queue.append([x, y])
    sheep = 0
    wolf = 0
    
    if graph[x][y] == 'o':
        sheep+=1
    elif graph[x][y] == 'v':
        wolf+=1
        
    graph[x][y] = '#'
    
    while queue:
        queueX, queueY = queue.popleft()

        for idx in range(4):
            newX, newY = queueX + dx[idx], queueY + dy[idx]

            if newX < 0 or newX >= r or newY < 0 or newY >= c or graph[newX][newY] == '#': 
                continue

            if graph[newX][newY] != '#': 
                if graph[newX][newY] == 'o':
                    sheep+=1
                elif graph[newX][newY] == 'v':
                    wolf+=1
        
                graph[newX][newY] = '#'
                queue.append([newX, newY])

    if sheep > wolf and (sheep + wolf != 0):
        sheep_count+=sheep
    elif sheep <= wolf and (sheep + wolf != 0):
        wolf_count+=wolf

for i in range(r):
    for j in range(c):
        if graph[i][j] != '#':
            bfs(i, j)
    
print(sheep_count, wolf_count)
```
